enslave_interaction = {
	desc = enslave_interaction.desc
	category = interaction_category_prison
	common_interaction = yes

	is_shown = {
		scope:recipient = {
			is_imprisoned_by = scope:actor
			is_ruler = no
			is_landed = no
			NOT = { has_trait = slave }
		}
	}
	auto_accept = yes

	on_accept = {
		enslave_effect = { SLAVE = scope:recipient OWNER = scope:actor }
	}

	ai_targets = { ai_recipients = prisoners }
	ai_frequency = 12
	ai_potential = { is_at_war = no }

	ai_will_do = {
		base = 0

		ai_value_modifier = {
			max = 5
			ai_compassion = -1
		}
		modifier = {
			add = -25
			scope:actor.dynasty = scope:recipient.dynasty
		}
		modifier = {
			add = -25
			is_close_or_extended_family_of = scope:recipient
		}
		modifier = {
			add = -25
			is_close_family_of = scope:recipient
		}
		modifier = {
			add = -25
			any_dynasty_member = {
				primary_heir = scope:recipient
			}
		}
		modifier = {
			add = 50
			has_execute_reason = scope:recipient
			OR = {
				has_trait = sadistic
				has_trait = lunatic
			}
		}
	}
}
